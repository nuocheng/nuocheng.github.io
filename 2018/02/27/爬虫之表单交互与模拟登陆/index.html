url=”https://www.douban.com/accounts/login“

params={
‘source’:’index_nav’,
‘form_email’:’‘,
‘form_password’:’*‘
}

print(html.text)

import requests
url=’https://www.douban.com‘
headers={
‘Cookies’:’*‘

}

html=requests.get(url,headers=headers)

print(html.text)
